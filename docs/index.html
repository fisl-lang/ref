Grammar

program   ::= ((" ")* line "\n")*
line      ::= statement | comment
comment   ::= "#" .*
statement ::=
      "let" var "be" expr
    | "let" var "be" expr "plus"  expr
    | "let" var "be" expr "minus" expr
    | "print" expr
    | "label" iden
    | "goto"  iden
    | "if" expr "equal"   expr "goto" iden 
    | "if" expr "unequal" expr "goto" iden 
    | "if" expr "greater" expr "goto" iden 
    | "if" expr "lesser"  expr "goto" iden 
    | "push" expr
    | "pull" expr
    | "call" iden
    | "return"
    | "write" expr "into" expr
    | "read"  var  "into" expr
    | "output" "newline"
    | "output" expr
    | "input"  var
    | "allocate" number "words" "for" var
    | "string"   string "into" var

expr ::= var | number
string :: = '"' .* '"'
var ::= iden
